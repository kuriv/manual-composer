# 配置文件

要开始在你的项目中使用 Composer ，只需要一个 `composer.json` 文件。

```json
{
    "name": "kuriv/composer",
    "description": "Composer 手册",
    "type": "project",
    "keywords": [
        "composer"
    ],
    "homepage": "http://kuriv.github.io/manual-composer/book/",
    "time": "2019-04-15 00:00:00",
    "license": "MIT",
    "authors": [
        {
            "name": "Kuriv Vesti",
            "email": "kuri.personal@gmail.com"
        }
    ],
    "support": {
        "email": "kuri.personal@gmail.com"
    },
    "require": {
        "php": ">=7.3.0"
    },
    "require-dev": {},
    "conflict": {},
    "replace": {},
    "provide": {},
    "suggest": {},
    "autoload": {
        "psr-4": {},
        "psr-0": {},
        "classmap": [],
        "files": []
    },
    "autoload-dev": {
        "psr-4": {},
        "psr-0": {},
        "classmap": [],
        "files": []
    },
    "minimum-stability": "stable",
    "prefer-stable": true,
    "config": {
        "process-timeout": 300,
        "use-include-path": false,
        "preferred-install": "auto",
        "github-protocols": [
            "git",
            "https",
            "ssh"
        ],
        "vendor-dir": "vendor",
        "bin-dir": "vendor/bin",
        "prepend-autoloader": true,
        "optimize-autoloader": false
    },
    "scripts": {},
    "extra": {},
    "bin": [],
    "archive": {}
}
```
